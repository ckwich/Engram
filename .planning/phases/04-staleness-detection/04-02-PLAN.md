```python
def api_stale():
    """Return stale memories list. Optional query params: days (int), type (time|code|all)."""
    days = request.args.get("days", None, type=int)
    filter_type = request.args.get("type", "all")
    if filter_type not in ("time", "code", "all"):
        filter_type = "all"
    results = memory_manager.get_stale_memories(days=days, type=filter_type)
    return jsonify(results)
```

```python
@app.route("/api/memory/<path:key>/reviewed", methods=["POST"])
def api_reviewed(key):
    """
    Mark a memory as reviewed without deleting it (STAL-04).
    - time-stale: resets last_accessed to now
    - code-stale: clears potentially_stale flag
    - both: applies both resets
    Reads stale_type from request JSON body: {"stale_type": "time"|"code"|"both"}
    """
    body = request.get_json(silent=True) or {}
    stale_type = body.get("stale_type", "both")

    data = memory_manager._load_json(key)
    if data is None:
        return jsonify({"error": "Memory not found"}), 404

    now = _now()
    if stale_type in ("time", "both"):
        data["last_accessed"] = now
    if stale_type in ("code", "both"):
        data["potentially_stale"] = False
        data["stale_reason"] = ""
        data["stale_flagged_at"] = None

    try:
        memory_manager._save_json(data)
    except Exception as e:
        return jsonify({"error": str(e)}), 503

    return jsonify({"reviewed": True, "key": key})
```

```javascript
// ── Stale Tab ────────────────────────────────────────────────────────────────
function toggleStaleTab() {
  staleTabActive = !staleTabActive;
  const btn = document.getElementById('btn-stale-tab');
  const stalePanel = document.getElementById('stale-panel');
  const memoryContainer = document.getElementById('memory-container');
  const searchResults = document.getElementById('search-results');

  if (staleTabActive) {
    btn.classList.add('active');
    stalePanel.classList.remove('hidden');
    memoryContainer.classList.add('hidden');
    searchResults.classList.add('hidden');
    loadStaleTab();
  } else {
    btn.classList.remove('active');
    stalePanel.classList.add('hidden');
    memoryContainer.classList.remove('hidden');
  }
}

async function loadStaleTab() {
  const listEl = document.getElementById('stale-list');
  const filterType = document.getElementById('stale-type-filter').value;
  listEl.innerHTML = '<div class="text-muted" style="padding:1rem;">Loading stale memories...</div>';

  try {
    const res = await fetch(`/api/stale?type=${filterType}`);
    const items = await res.json();

    if (!items.length) {
      listEl.innerHTML = '<div class="empty-state"><div class="icon">&#x2705;</div><p>No stale memories found.</p></div>';
      return;
    }

    listEl.innerHTML = items.map(m => {
      const badge = {
        time: '<span class="stale-badge stale-time">Time stale</span>',
        code: '<span class="stale-badge stale-code">Code changed</span>',
        both: '<span class="stale-badge stale-both">Time + Code</span>',
      }[m.stale_type] || '';
      const tags = (m.tags || []).map(t => `<span class="tag-chip">${esc(t)}</span>`).join('');
      return `
        <div class="stale-row" data-key="${esc(m.key)}" data-stale-type="${esc(m.stale_type)}">
          <div class="stale-row-header">
            ${badge}
            <span class="stale-title">${esc(m.title)}</span>
            <button class="btn btn-sm stale-reviewed-btn" onclick="markReviewed('${esc(m.key)}', '${esc(m.stale_type)}', this)">Mark Reviewed</button>
          </div>
          <div class="stale-row-meta">
            <span class="text-muted" style="font-size:0.8rem;">${esc(m.key)}</span>
            <span class="stale-detail">${esc(m.stale_detail)}</span>
          </div>
          <div class="tags">${tags}</div>
        </div>
      `;
    }).join('');
  } catch (err) {
    listEl.innerHTML = `<div class="text-muted" style="padding:1rem;">Error loading stale memories: ${esc(err.message)}</div>`;
  }
}

async function markReviewed(key, staleType, btn) {
  btn.disabled = true;
  btn.textContent = 'Saving...';
  try {
    const res = await fetch(`/api/memory/${encodeURIComponent(key)}/reviewed`, {
      method: 'POST',
      headers: {'Content-Type': 'application/json'},
      body: JSON.stringify({stale_type: staleType}),
    });
    if (!res.ok) throw new Error(`HTTP ${res.status}`);
    // Remove row from list with fade
    const row = btn.closest('.stale-row');
    row.style.opacity = '0';
    row.style.transition = 'opacity 0.3s';
    setTimeout(() => row.remove(), 300);
  } catch (err) {
    btn.textContent = 'Error';
    btn.disabled = false;
  }
}
```
